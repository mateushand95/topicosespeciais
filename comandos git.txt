git --version = verifica a versao do git
git init = inicia um repositorio vazio na pasta atual
git config --global user.name "Bunno SousA Monte"
git config --global user.email "brunnosousa.bs97@gmail.com"
git status = verifica tudo que nao foi salvo
git add . = colocar os arquivos na area de transferencia
git commit -m "mensagem" = falar uma mensagem para indentificar um commit
git log = mostra todos os log dos comits realizados
git remoto add origin link do repositorio = vincular a pasta local repositorio
git push -u origin master = fazer commit pela primeira vez
